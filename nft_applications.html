NFT 开发指南

介绍

NFT（Non-Fungible Token，非同质化代币）是区块链技术中的一种独特资产，每个NFT都有唯一的标识符，不可互换。本指南将帮助你开始NFT和Web3开发之旅。

智能合约开发

+

NFT合约示例

MyNFT.sol 

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/utils/Counters.sol";

contract MyNFT is ERC721 {
    using Counters for Counters.Counter;
    Counters.Counter private _tokenIds;

    constructor() ERC721("MyNFT", "MNFT") {}

    function mintNFT(address recipient, string memory tokenURI)
        public returns (uint256)
    {
        _tokenIds.increment();
        uint256 newItemId = _tokenIds.current();
        _mint(recipient, newItemId);
        _setTokenURI(newItemId, tokenURI);
        return newItemId;
    }
}

import Web3 from 'web3';
import MyNFT from '../build/contracts/MyNFT.json';

const web3 = new Web3(Web3.givenProvider || "http://localhost:7545");
const contractAddress = MyNFT.networks[5777].address;
const contract = new web3.eth.Contract(MyNFT.abi, contractAddress);

export { web3, contract };
